74 void QDiscordRestComponent::sendMessage(const QString& content, QDiscordChannel* channel, bool tts)
75 {
76  if(_authentication == "")
77  return;
78  if(!channel)
79  return;
80  QString id = channel->id();
81  QJsonObject object;
82  object["content"] = content;
83  int seed = QDateTime::currentDateTime().toTime_t();
84  std::minstd_rand0 generator(seed);
85  int nonce;
86  while(true)
87  {
88  nonce = (int)generator();
89  for(int i = 0; i < _messageSendQueue.length(); i++)
90  if(_messageSendQueue[i].nonce == nonce)
91  continue;
92  break;
93  }
94  object["nonce"] = nonce;
95  if(tts)
96  object["tts"] = true;
97  //MessageParams params = {nonce, content, channel, tts};
98  //_messageSendQueue.append(params);
99  post(object, QUrl(QString(QDiscordUtilities::endPoints.channels + "/" + id + "/messages")),
100  std::function<void ()>([=](){
101  QNetworkReply* reply = static_cast<QNetworkReply*>(sender());
102  if(!reply)
103  return;
104  if(reply->error() != QNetworkReply::NoError)
105  emit messageSendFailed(reply->error());
106  else
107  {
108  QJsonDocument document = QJsonDocument::fromJson(reply->readAll());
109  QJsonObject object = document.object();
110  QDiscordMessage message(object, nullptr);
111  emit messageSent(message);
112  }
113  reply->deleteLater();
114  }));
115 }

161 void QDiscordRestComponent::post(const QJsonObject& object, const QUrl& url, std::function<void()> function)
162 {
163  QString userAgent = "DiscordBot (" + QDiscordUtilities::libLink +
164  ", v" + QDiscordUtilities::libMajor + ":" + QDiscordUtilities::libMinor + ")" +
165  "; " + QDiscordUtilities::botName;
166  QJsonDocument document;
167  document.setObject(object);
168  QNetworkRequest request(url);
169  if(_authentication != "")
170  request.setRawHeader("Authorization", _authentication.toUtf8());
171  request.setRawHeader("User-Agent", userAgent.toUtf8());
172  request.setRawHeader("content-type", "application/json");
173  QNetworkReply* reply = _manager.post(request, document.toJson(QJsonDocument::Compact));
174  connect(reply, &QNetworkReply::finished, this, function);
175  qDebug()<<this<<"posted: "<<object<<" to "<<url;
176 }
