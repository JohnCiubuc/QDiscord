32 void QDiscordRestComponent::login(const QString& email, const QString& password)
33 {
34  QJsonObject object;
35  object["email"] = email;
36  object["password"] = password;
37  post(object, QDiscordUtilities::endPoints.login,
38  [=](){
39  QNetworkReply* reply = static_cast<QNetworkReply*>(sender());
40  if(!reply)
41  return;
42  if(reply->error() != QNetworkReply::NoError)
43  emit loginFailed(reply->error());
44  else
45  {
46  _authentication =
47  QJsonDocument::fromJson(
48  reply->readAll()
49  ).object().value("token").toString();
50  emit tokenVerified(_authentication, QDiscordTokenType::None);
51  }
52  reply->deleteLater();
53  });
54 }
