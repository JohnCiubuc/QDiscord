63 QString QDiscordUtilities::networkErrorToString(QNetworkReply::NetworkError error)
64 {
65  switch((int)error)
66  {
67  case 200:
68  return "200 (OK): The response completed sucessfully.";
69  case 201:
70  return "200 (CREATED): The entity was created successfully.";
71  case 304:
72  return "304 (NOT MODIFIED): The entity was not modified (no action was taken).";
73  case 400:
74  return "400 (BAD REQUEST): The request was improperly formatted, or the server couldn't understand it.";
75  case 401:
76  return "401 (UNAUTHORIZED): The Authorization header was missing or invalid.";
77  case 403:
78  return "403 (FORBIDDEN): The Authorization token you passed did not have permission to the resource";
79  case 404:
80  return "404 (NOT FOUND): The resource at the location specified doesn't exist.";
81  case 405:
82  return "405 (METHOD NOT ALLOWED): The HTTP method used is not valid for the location specified.";
83  case 429:
84  return "429 (TOO MANY REQUESTS): You've made too many requests.";
85  case 502:
86  return "502 (GATEWAY UNAVAILABLE): There was not a gateway available to process your request. Wait a bit and retry.";
87  default:
88  if((int)error > 499 && (int)error < 600)
89  return "5xx (SERVER ERROR): The server had an error processing your request.";
90  else return "xxx (UNKNOWN): Unknown error.";
91  }
92 }
