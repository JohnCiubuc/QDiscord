48 QString QDiscordUtilities::networkErrorToString(QNetworkReply::NetworkError error)
49 {
50  switch((int)error)
51  {
52  case 200:
53  return "200 (OK): The response completed sucessfully.";
54  case 201:
55  return "200 (CREATED): The entity was created successfully.";
56  case 304:
57  return "304 (NOT MODIFIED): The entity was not modified (no action was taken).";
58  case 400:
59  return "400 (BAD REQUEST): The request was improperly formatted, or the server couldn't understand it.";
60  case 401:
61  return "401 (UNAUTHORIZED): The Authorization header was missing or invalid.";
62  case 403:
63  return "403 (FORBIDDEN): The Authorization token you passed did not have permission to the resource";
64  case 404:
65  return "404 (NOT FOUND): The resource at the location specified doesn't exist.";
66  case 405:
67  return "405 (METHOD NOT ALLOWED): The HTTP method used is not valid for the location specified.";
68  case 429:
69  return "429 (TOO MANY REQUESTS): You've made too many requests.";
70  case 502:
71  return "502 (GATEWAY UNAVAILABLE): There was not a gateway available to process your request. Wait a bit and retry.";
72  default:
73  if((int)error > 499 && (int)error < 600)
74  return "5xx (SERVER ERROR): The server had an error processing your request.";
75  else return "xxx (UNKNOWN): Unknown error.";
76  }
77 }
