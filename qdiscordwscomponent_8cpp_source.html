91 void QDiscordWsComponent::setStatus(bool idle, QDiscordGame game)
92 {
93  if(_token == "")
94  return;
95  if(_gateway == "")
96  return;
97  QJsonDocument document;
98  QJsonObject object;
99  object["op"] = 3;
100  QJsonObject presenceObject;
101  presenceObject["idle_since"] = idle ?
102  QDateTime::currentMSecsSinceEpoch() : QJsonValue();
103  if(game.name() != "")
104  {
105  QJsonObject gameObject;
106  gameObject["name"] = game.name();
107  gameObject["url"] = game.url()==""?QJsonValue():game.url();
108  gameObject["type"] = static_cast<int>(game.type());
109  presenceObject["game"] = gameObject;
110  }
111  else
112  presenceObject["game"] = QJsonValue();
113  object["d"] = presenceObject;
114  document.setObject(object);
115  _socket.sendTextMessage(document.toJson(QJsonDocument::Compact));
116 }

118 void QDiscordWsComponent::login(const QString& token,
119  QDiscordTokenType tokenType)
120 {
121  if(_reconnectTimer.isActive())
122  _reconnectTimer.stop();
123  QJsonDocument document;
124  QJsonObject mainObject;
125  mainObject["op"] = 2;
126  QJsonObject dataObject;
127  dataObject["token"] =
128  QDiscordUtilities::convertTokenToType(token, tokenType);
129  dataObject["v"] = 5;
130  dataObject["properties"] =
131  QJsonObject({
132  {"$os", QSysInfo::kernelType()},
133  {"$browser", QDiscordUtilities::libName},
134  {"$device", QDiscordUtilities::libName},
135  {"$referrer", "https://discordapp.com/@me"},
136  {"$referring_domain", "discordapp.com"}
137  });
138  dataObject["large_threshold"] = 100;
139  dataObject["compress"] = false;
140  mainObject["d"] = dataObject;
141  document.setObject(mainObject);
142  _socket.sendTextMessage(document.toJson(QJsonDocument::Compact));
143 }

145 void QDiscordWsComponent::reconnect()
146 {
147  if(_reconnectTimer.isActive())
148  _reconnectTimer.stop();
149  if(_token == "")
150  return;
151  if(_gateway == "")
152  return;
153 
154 #ifdef QDISCORD_LIBRARY_DEBUG
155  qDebug()<<this<<"reconnecting";
156 #endif
157 
158  if(_reconnectAttempts > _maxReconnectAttempts &&
159  _maxReconnectAttempts != -1)
160  {
161 #ifdef QDISCORD_LIBRARY_DEBUG
162  qDebug()<<"maximum reconnect attempts reached, stopping";
163 #endif
164 
165  _reconnectAttempts = 0;
166  close();
167  emit reconnectImpossible();
168  }
169  else
170  {
171  if(_maxReconnectAttempts != -1)
172  _reconnectAttempts++;
173  emit attemptingReconnect();
174  _socket.open(_gateway);
175  }
176 }

229 void QDiscordWsComponent::textMessageReceived(const QString& message)
230 {
231  QJsonDocument document = QJsonDocument::fromJson(message.toUtf8());
232  QJsonObject object = document.object();
233  if(_useDumpfile)
234  {
235  QFile file("DUMPFILE.txt");
236  file.open(QFile::WriteOnly|QFile::Append);
237  file.write(document.toJson(QJsonDocument::Indented));
238  file.flush();
239  file.close();
240  }
241 
242 #ifdef QDISCORD_LIBRARY_DEBUG
243  qDebug()<<this<<"op:"<<object["op"].toInt()<<" t:"<<object["t"].toString();
244 #endif
245 
246  switch(object["op"].toInt(-1))
247  {
248  case 0:
249  if(_eventDispatchTable.keys().contains(object["t"].toString()))
250  _eventDispatchTable[object["t"].toString()](object["d"].toObject());
251 #ifdef QDISCORD_LIBRARY_DEBUG
252  else
253  qDebug()<<this<<"encountered an event not in the dispatch table";
254 #endif
255  break;
256  case -1:
257 #ifdef QDISCORD_LIBRARY_DEBUG
258  qDebug()<<this<<"error while parsing operation code";
259 #endif
260  break;
261  default:
262 #ifdef QDISCORD_LIBRARY_DEBUG
263  qDebug()<<this<<"encountered an unhandled operation";
264 #else
265  ;
266 #endif
267  }
268 }
