22 QDiscordGuild::QDiscordGuild(const QJsonObject& object)
23 {
24  _id = object["id"].toString("");
25  _unavailable = object["unavailable"].toBool(false);
26  _name = object["name"].toString("");
27  _verificationLevel = object["verification_level"].toInt(0);
28  _afkTimeout = object["afk_timeout"].toInt(0);
29  _memberCount = object["member_count"].toInt(1);
30  _joinedAt = QDateTime::fromString(object["joined_at"].toString(""), Qt::ISODate);
31  QJsonArray memberArray = object["members"].toArray();
32  for(int i = 0; i < memberArray.count(); i++)
33  {
34  QDiscordMember* member = new QDiscordMember(memberArray[i].toObject(), this);
35  if(_members.keys().contains(member->user()->id()))
36  {
37  delete _members.value(member->user()->id());
38  _members.insert(member->user()->id(), member);
39  }
40  else
41  _members.insert(member->user()->id(), member);
42  }
43  QJsonArray channelArray = object["channels"].toArray();
44  for(int i = 0; i < channelArray.count(); i++)
45  {
46  QDiscordChannel* channel = new QDiscordChannel(channelArray[i].toObject(), this);
47  if(_channels.keys().contains(channel->id()))
48  {
49  delete _channels.value(channel->id());
50  _channels.insert(channel->id(), channel);
51  }
52  else
53  _channels.insert(channel->id(), channel);
54  }
55 
56  if(QDiscordUtilities::debugMode)
57  qDebug()<<"QDiscordGuild("<<this<<") constructed";
58 }
