68 QDiscordMessage::QDiscordMessage(const QDiscordMessage& other)
69 {
70  _id = other.id();

74  _channel = other.channel();

76  _tts = other.tts();

78  QMap<QDiscordUser*, bool> otherMentions = other.mentionsWithOwnership();
79  for(QDiscordUser* item : otherMentions.keys())
80  {
81  if(otherMentions.value(item))
82  _mentions.insert(new QDiscordUser(*item), true);
83  else
84  _mentions.insert(item, false);
85  }
86 }
