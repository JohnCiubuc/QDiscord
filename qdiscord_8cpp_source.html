99  connect(&_ws, &QDiscordWsComponent::loginSuccess, this, &QDiscord::loginSuccess);
100  connect(&_ws, &QDiscordWsComponent::disconnected, this, &QDiscord::disconnected);
101  connect(&_ws, &QDiscordWsComponent::error, this, &QDiscord::disconnected);
