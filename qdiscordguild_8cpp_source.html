21 QDiscordGuild::QDiscordGuild(const QJsonObject& object)
22 {
23  _id = object["id"].toString("");
24  _unavailable = object["unavailable"].toBool(false);
25  _name = object["name"].toString("");
26  _verificationLevel = object["verification_level"].toInt(0);
27  _afkTimeout = object["afk_timeout"].toInt(0);
28  _memberCount = object["member_count"].toInt(1);
29  _joinedAt = QDateTime::fromString(object["joined_at"].toString(""), Qt::ISODate);
30  QJsonArray memberArray = object["members"].toArray();
31  for(int i = 0; i < memberArray.count(); i++)
32  {
33  QDiscordMember* member = new QDiscordMember(memberArray[i].toObject(), this);
34  if(_members.keys().contains(member->user()->id()))
35  {
36  delete _members.value(member->user()->id());
37  _members.insert(member->user()->id(), member);
38  }
39  else
40  _members.insert(member->user()->id(), member);
41  }
42  QJsonArray channelArray = object["channels"].toArray();
43  for(int i = 0; i < channelArray.count(); i++)
44  {
45  QDiscordChannel* channel = new QDiscordChannel(channelArray[i].toObject(), this);
46  if(_channels.keys().contains(channel->id()))
47  {
48  delete _channels.value(channel->id());
49  _channels.insert(channel->id(), channel);
50  }
51  else
52  _channels.insert(channel->id(), channel);
53  }
54  qDebug()<<"QDiscordGuild("<<this<<") constructed";
55 }
