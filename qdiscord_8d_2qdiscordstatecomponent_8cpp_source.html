55 void QDiscordStateComponent::readyReceived(const QJsonObject& object)
56 {
57  if(_self)
58  {
59  delete _self;
60  _self = nullptr;
61  }
62  _self = new QDiscordUser(object["user"].toObject());
63  emit selfCreated(*_self);
64  QJsonArray guildArray = object["guilds"].toArray();
65  for(int i = 0; i < guildArray.count(); i++)
66  guildCreateReceived(guildArray[i].toObject());
67  QJsonArray privateMessageArray = object["private_channels"].toArray();
68  for(int i = 0; i < privateMessageArray.count(); i++)
69  channelCreateReceived(privateMessageArray[i].toObject());
70 }

121 void QDiscordStateComponent::guildMemberRemoveReceived(const QJsonObject& object)
122 {
123  QDiscordGuild* guildPtr = guild(object["guild_id"].toString(""));
124  QDiscordMember* member;
125  if(guildPtr)
126  {
127  QDiscordMember* tmpMember = guildPtr->member(object["user"].toObject()["id"].toString(""));
128  if(tmpMember)
129  {
130  member = new QDiscordMember(*tmpMember);
131  guildPtr->removeMember(tmpMember);
132  }
133  else
134  member = new QDiscordMember(object, nullptr);
135  }
136  else
137  member = new QDiscordMember(object, nullptr);
138  QDiscordMember newMember(*member);
139  emit guildMemberRemoved(newMember);
140  delete member;
141 }

224 void QDiscordStateComponent::channelCreateReceived(const QJsonObject& object)
225 {
226  QDiscordChannel* channel = new QDiscordChannel(object, guild(object["guild_id"].toString("")));
227  if(channel->isPrivate())
228  {
229  if(_privateChannels.keys().contains(channel->id()))
230  {
231  delete _privateChannels.value(channel->id());
232  _privateChannels.insert(channel->id(), channel);
233  }
234  else
235  _privateChannels.insert(channel->id(), channel);
236  emit privateChannelCreated(channel);
237  }
238  else
239  {
240  if(!channel->guild())
241  {
242  delete channel;
243  return;
244  }
245  channel->guild()->addChannel(channel);
246  emit channelCreated(channel);
