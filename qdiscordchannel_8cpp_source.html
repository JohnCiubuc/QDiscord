23 {

26  _lastMessageId = object["last_message_id"].toString("");
27  _name = object["name"].toString("");
28  _position = object["position"].toInt(0);
29  _topic = object["topic"].toString("");
30  QString type = object["type"].toString("text");
31  if(type == "text")
32  _type = ChannelType::Text;
33  else if(type == "voice")
34  _type = ChannelType::Voice;
35  else
36  _type = ChannelType::UnknownType;
37  _guild = guild;
38  QJsonObject recipientObject = object["recipient"].toObject();
39  _recipient = nullptr;
40  if(_isPrivate)
